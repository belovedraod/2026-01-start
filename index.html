        function loadBookCover(imgElement, book) {

            let initialUrl;

            if (book.google_id) {

                initialUrl = `https://books.google.com/books/content?id=${book.google_id}&printsec=frontcover&img=1&zoom=1`;

            } else {

                initialUrl = `https://covers.openlibrary.org/b/isbn/${book.isbn_en}-L.jpg`;

            }

            const fallbackUrl = `https://covers.openlibrary.org/b/isbn/${book.isbn_en}-L.jpg`;

            const textPlaceholderUrl = `https://placehold.co/400x600/f5f5f4/a8a29e?text=${encodeURIComponent(book.title.substring(0,10))}`;


            imgElement.src = initialUrl;

            imgElement.onload = function() {

                if (this.naturalWidth < 5 && this.src.includes('google')) {

                    if (!this.getAttribute('data-tried-fallback')) {

                        this.setAttribute('data-tried-fallback', 'true');

                        this.src = fallbackUrl;

                    } else {

                         this.src = textPlaceholderUrl;

                    }

                } else {

                    this.classList.add('loaded');

                }

            };

            imgElement.onerror = function() {

                if (!this.getAttribute('data-tried-fallback')) {

                    this.setAttribute('data-tried-fallback', 'true');

                    this.src = fallbackUrl;

                } else {

                    this.src = textPlaceholderUrl;

                }

            };

        }
